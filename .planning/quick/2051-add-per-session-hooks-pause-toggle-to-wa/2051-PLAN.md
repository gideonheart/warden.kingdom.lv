async function readPauseStateMap(): Promise<Record<string, { paused: boolean; updatedAt: string }>> {
  try {
    const rawContent = await readFile(PAUSE_STATE_FILE_PATH, 'utf-8');
    return JSON.parse(rawContent);

  }
}

- Poll `GET /api/gsd/hooks-pause-state` every 10 seconds (same interval as `useGsdRegistry`)
- Return a `Record<string, boolean>` mapping session names to paused state (extract just the `paused` boolean from each entry)
- On fetch error, return previous state (fail-open, no error UI needed — pause is informational)
- Export `useHooksPauseState()` returning `{ pauseMap: Record<string, boolean>, isLoading: boolean }`

Why a new hook instead of extending useAgentStateFiles: useAgentStateFiles makes N per-session requests to `/api/gsd/sessions/:session/state` — pause state is a single file read, different data source, different polling cadence. Keeping them separate is cleaner SRP.

3. Add optimistic toggle state (same pattern as `toggleEnabled` in `useGsdRegistry`):
```typescript
const [optimisticHooksPaused, setOptimisticHooksPaused] = useState<Record<string, boolean>>({});

const toggleHooksPaused = useCallback(async (sessionName: string, currentPaused: boolean) => {
  const newPaused = !currentPaused;
  setOptimisticHooksPaused((previous) => ({ ...previous, [sessionName]: newPaused }));
  try {
    const response = await fetch(`/api/gsd/sessions/${sessionName}/hooks-paused`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ paused: newPaused }),
    });
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    setOptimisticHooksPaused((previous) => {
      const next = { ...previous };
      delete next[sessionName];
      return next;
    });
  } catch {
    setOptimisticHooksPaused((previous) => ({ ...previous, [sessionName]: currentPaused }));
  }
}, []);
```
